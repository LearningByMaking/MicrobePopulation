;ulogo file for $projectName

 constants 
[
  	[rpacket $1fc0]
  	[minTemp 500]
  	[maxTemp 550]
]


 to ul-power
	 init-pins 
	 redon
	 wait 10
	 redoff
 end
 
 to ul-init 
	 redon	         
	 wait 10	 
	 redoff
     
 end
 
 to ul-go 
 	 redon
	 wait 10
	 redoff

	 loop
	 [
		redon	 	
		collect-data
		wait 10 
		redoff
	  	if(readADC 6 > maxTemp)[heaterOff]
          	if(readADC 6 < minTemp)[heaterOn]
	 	wait 100
	 ]
 
 end

to heaterOn
end

to heaterOff
end


 to collect-data
 	init-packet 55 ;55 represents pad and temp ADC reading	
 	fprint readADC 7 ;pads
        fprint readADC 6 ;temp 
 	add-checksum
 end
